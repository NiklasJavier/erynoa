    primary: "vehicle_availability"
    secondary:
      - "fuel_efficiency"
      - "maintenance_due"

      type: "geohash"
      precision: 5
      fallback: "region_code"

    informed:
      - name: "fleet_dispatch_score"
        formula: |
          0.4 * (1 - distance_to_task / max_range) +
          0.3 * vehicle.fuel_level +
          0.2 * driver.trust_score +
          0.1 * (1 - maintenance_urgency)

      - name: "energy_efficiency_route"
        formula: |
          route_elevation_gain * 0.3 +
          traffic_density * 0.4 +
          weather_resistance * 0.3

      - "bfs_nearest_vehicle"
      - "dfs_driver_chain"

    hard:

      - "vehicle.last_maintenance < 30_days" # preference
