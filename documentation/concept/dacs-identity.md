### 10.1 gRPC API

```protobuf
syntax = "proto3";

package erynoa.dacs.v1;

service DACSService {
  // DID Operations
  rpc CreateDID(CreateDIDRequest) returns (CreateDIDResponse);
  rpc UpdateDID(UpdateDIDRequest) returns (UpdateDIDResponse);
  rpc DeactivateDID(DeactivateDIDRequest) returns (DeactivateDIDResponse);
  rpc ResolveDID(ResolveDIDRequest) returns (ResolveDIDResponse);

  // Verifiable Credentials
  rpc IssueVC(IssueVCRequest) returns (IssueVCResponse);
  rpc VerifyVC(VerifyVCRequest) returns (VerifyVCResponse);
  rpc RevokeVC(RevokeVCRequest) returns (RevokeVCResponse);

  // Anchor Verification
  rpc GetAnchorProof(GetAnchorProofRequest) returns (GetAnchorProofResponse);
  rpc VerifyAnchor(VerifyAnchorRequest) returns (VerifyAnchorResponse);

  // Registry
  rpc GetRegistryInfo(GetRegistryInfoRequest) returns (GetRegistryInfoResponse);
}

message CreateDIDRequest {
  bytes public_key = 1;
  string key_type = 2;
  repeated Service services = 3;
  bytes signature = 4;
}

message CreateDIDResponse {
  string did = 1;
  DIDDocument document = 2;
  repeated AnchorProof proofs = 3;
}

message AnchorProof {
  string chain = 1;
  bytes tx_hash = 2;
  uint64 block_number = 3;
  uint64 timestamp = 4;
  bytes threshold_signature = 5;
}
```
